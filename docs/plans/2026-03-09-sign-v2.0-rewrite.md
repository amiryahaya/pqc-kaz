# KAZ-SIGN v2.0 Rewrite Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Rewrite the C KAZ-SIGN library to match Java kaz-pqc-core-v2.0's 3-component signature scheme (S1, S2, S3) exactly.

**Architecture:** Replace the current 2-component scheme with RSA-like parameters (N, phiN, g1=65537, g2=65539). SHA-256 only for all levels. New OID arc. Clean break from v1 — all keys, signatures, KAT vectors invalidated.

### Task 3: Rewrite `sign.c` — Hashing Function

**Files:**
- Modify: `SIGN/src/internal/sign.c`

**Step 1: Rewrite `kaz_sign_hash_ex()`**

Always use SHA-256 regardless of level, then zero-pad to `params->hash_bytes`:

```c
int kaz_sign_hash_ex(kaz_sign_level_t level,
                     const unsigned char *msg,
                     unsigned long long msglen,
                     unsigned char *hash)
{
    const kaz_sign_level_params_t *params;
    EVP_MD_CTX *ctx = NULL;
    unsigned char sha256_digest[32];
    unsigned int digest_len = 0;
    int ret = KAZ_SIGN_ERROR_HASH;

    params = kaz_sign_get_level_params(level);
    if (!params) return KAZ_SIGN_ERROR_INVALID;
    if (hash == NULL) return KAZ_SIGN_ERROR_INVALID;
    if (msg == NULL && msglen > 0) return KAZ_SIGN_ERROR_INVALID;

    /* Always SHA-256 */
    ctx = EVP_MD_CTX_new();
    if (!ctx) return KAZ_SIGN_ERROR_MEMORY;

    if (EVP_DigestInit_ex(ctx, EVP_sha256(), NULL) != 1) goto cleanup;
    if (msglen > 0 && EVP_DigestUpdate(ctx, msg, (size_t)msglen) != 1) goto cleanup;
    if (EVP_DigestFinal_ex(ctx, sha256_digest, &digest_len) != 1) goto cleanup;

    /* Zero-pad to hash_bytes (32/48/64) */
    memset(hash, 0, params->hash_bytes);
    memcpy(hash, sha256_digest, digest_len < params->hash_bytes ? digest_len : params->hash_bytes);

    ret = KAZ_SIGN_SUCCESS;

cleanup:
    EVP_MD_CTX_free(ctx);
    return ret;
}
```

**Step 2: Remove the old `hash_md` field from runtime params**

The old code cached `EVP_sha256()` / `EVP_sha384()` / `EVP_sha512()` per level. Remove that — we always use `EVP_sha256()`.

### Task 4: Rewrite `sign.c` — Key Generation

**Files:**
- Modify: `SIGN/src/internal/sign.c`

**Step 1: Write helper `sample_in_range()`**

Matches Java `getRandom.getRandomIntInRange()`:

```c
/* Sample a random BIGNUM uniformly in [lower, upper] */
static int sample_in_range(BIGNUM *result, const BIGNUM *lower,
                           const BIGNUM *upper, BN_CTX *ctx)
{
    BIGNUM *range = NULL, *r = NULL;
    int bits, ret = 0;

    range = BN_new();
    r = BN_new();
    if (!range || !r) goto fail;

    /* range = upper - lower + 1 */
    if (!BN_sub(range, upper, lower)) goto fail;
    if (!BN_add_word(range, 1)) goto fail;
    bits = BN_num_bits(range);

    /* Rejection sampling */
    do {
        if (!BN_rand(r, bits, BN_RAND_TOP_ANY, BN_RAND_BOTTOM_ANY)) goto fail;
    } while (BN_cmp(r, range) >= 0);

    /* result = r + lower */
    if (!BN_add(result, r, lower)) goto fail;
    ret = 1;

fail:
    BN_free(range);
    BN_free(r);
    return ret;
}
```

**Step 2: Rewrite `kaz_sign_keypair_ex()`**

Match Java `KAZSIGNKeyGenerator.generateKeyPair()`:

```c
int kaz_sign_keypair_ex(kaz_sign_level_t level,
                        unsigned char *pk,
                        unsigned char *sk)
{
    const kaz_sign_level_params_t *params;
    kaz_runtime_params_t *rp;
    BN_CTX *ctx = NULL;
    BIGNUM *s = NULL, *t = NULL, *v = NULL;
    BIGNUM *lower_s = NULL, *lower_t = NULL;
    BIGNUM *tmp1 = NULL, *tmp2 = NULL;
    int ret = KAZ_SIGN_ERROR_MEMORY;

    params = kaz_sign_get_level_params(level);
    if (!params) return KAZ_SIGN_ERROR_INVALID;
    if (!pk || !sk) return KAZ_SIGN_ERROR_INVALID;

    ret = kaz_sign_init_level(level);
    if (ret != KAZ_SIGN_SUCCESS) return ret;
    rp = get_runtime_params(level);
    if (!rp) return KAZ_SIGN_ERROR_INVALID;

    ctx = BN_CTX_new();
    s = BN_new(); t = BN_new(); v = BN_new();
    lower_s = BN_new(); lower_t = BN_new();
    tmp1 = BN_new(); tmp2 = BN_new();
    if (!ctx || !s || !t || !v || !lower_s || !lower_t || !tmp1 || !tmp2) goto cleanup;

    /* lower_s = 2^(lOg1N - 2) */
    if (!BN_set_word(lower_s, 1)) goto cleanup;
    if (!BN_lshift(lower_s, lower_s, rp->lOg1N - 2)) goto cleanup;

    /* lower_t = 2^(lOg2N - 2) */
    if (!BN_set_word(lower_t, 1)) goto cleanup;
    if (!BN_lshift(lower_t, lower_t, rp->lOg2N - 2)) goto cleanup;

    /* s = random in [2^(lOg1N-2), Og1N] */
    if (!sample_in_range(s, lower_s, rp->Og1N, ctx)) { ret = KAZ_SIGN_ERROR_RNG; goto cleanup; }

    /* t = random in [2^(lOg2N-2), Og2N] */
    if (!sample_in_range(t, lower_t, rp->Og2N, ctx)) { ret = KAZ_SIGN_ERROR_RNG; goto cleanup; }

    /* v = g1^s * g2^t mod N */
    if (!BN_mod_exp_mont_consttime(tmp1, rp->g1, s, rp->N, ctx)) goto cleanup;
    if (!BN_mod_exp_mont_consttime(tmp2, rp->g2, t, rp->N, ctx)) goto cleanup;
    if (!BN_mod_mul(v, tmp1, tmp2, rp->N, ctx)) goto cleanup;

    /* Serialize: pk = v (v_bytes), sk = s || t (s_bytes + t_bytes) */
    memset(pk, 0, params->public_key_bytes);
    memset(sk, 0, params->secret_key_bytes);
    BN_bn2binpad(v, pk, (int)params->v_bytes);
    BN_bn2binpad(s, sk, (int)params->s_bytes);
    BN_bn2binpad(t, sk + params->s_bytes, (int)params->t_bytes);

    ret = KAZ_SIGN_SUCCESS;

cleanup:
    BN_free(s); BN_free(t); BN_free(v);
    BN_free(lower_s); BN_free(lower_t);
    BN_free(tmp1); BN_free(tmp2);
    BN_CTX_free(ctx);
    return ret;
}
```

**Files:**

**Step 1: Rewrite `kaz_sign_signature_ex()`**

Match Java `KAZSIGNSigner.sign()`. Produces message-recovery format: S1 || S2 || S3 || message.

```c
int kaz_sign_signature_ex(kaz_sign_level_t level,
                           unsigned char *sig,
                           unsigned long long *siglen,
                           const unsigned char *msg,
                           unsigned long long msglen,
                           const unsigned char *sk)
{
    const kaz_sign_level_params_t *params;
    kaz_runtime_params_t *rp;
    BN_CTX *ctx = NULL;
    BIGNUM *s_key = NULL, *t_key = NULL;
    BIGNUM *h = NULL, *e1 = NULL, *e2 = NULL;
    BIGNUM *S1 = NULL, *S2 = NULL, *S3 = NULL;
    BIGNUM *tmp1 = NULL, *tmp2 = NULL, *tmp3 = NULL;
    BIGNUM *lower_e1 = NULL, *lower_e2 = NULL;
    BIGNUM *e1_inv = NULL;
    unsigned char *hash_buf = NULL;
    int ret = KAZ_SIGN_ERROR_MEMORY;

    params = kaz_sign_get_level_params(level);
    if (!params) return KAZ_SIGN_ERROR_INVALID;
    if (!sig || !siglen || !sk) return KAZ_SIGN_ERROR_INVALID;
    if (!msg && msglen > 0) return KAZ_SIGN_ERROR_INVALID;

    ret = kaz_sign_init_level(level);
    if (ret != KAZ_SIGN_SUCCESS) return ret;
    rp = get_runtime_params(level);

    /* Hash message: SHA-256, zero-padded to hash_bytes */
    hash_buf = calloc(1, params->hash_bytes);
    if (!hash_buf) return KAZ_SIGN_ERROR_MEMORY;

    ret = kaz_sign_hash_ex(level, msg, msglen, hash_buf);
    if (ret != KAZ_SIGN_SUCCESS) { free(hash_buf); return ret; }

    ctx = BN_CTX_new();
    s_key = BN_new(); t_key = BN_new();
    h = BN_new(); e1 = BN_new(); e2 = BN_new();
    S1 = BN_new(); S2 = BN_new(); S3 = BN_new();
    tmp1 = BN_new(); tmp2 = BN_new(); tmp3 = BN_new();
    lower_e1 = BN_new(); lower_e2 = BN_new();
    e1_inv = BN_new();
    /* ... NULL checks omitted for brevity, see full implementation ... */

    /* Deserialize secret key: s || t */
    BN_bin2bn(sk, (int)params->s_bytes, s_key);
    BN_bin2bn(sk + params->s_bytes, (int)params->t_bytes, t_key);
    BN_set_flags(s_key, BN_FLG_CONSTTIME);
    BN_set_flags(t_key, BN_FLG_CONSTTIME);

    /* h = BigInteger(1, hash_buf) */
    BN_bin2bn(hash_buf, (int)params->hash_bytes, h);

    /* lower_e1 = 2^(lOg1N - 2) */
    BN_set_word(lower_e1, 1);
    BN_lshift(lower_e1, lower_e1, rp->lOg1N - 2);
    /* lower_e2 = 2^(lOg2N - 2) */
    BN_set_word(lower_e2, 1);
    BN_lshift(lower_e2, lower_e2, rp->lOg2N - 2);

    /* e1 = sampleInRange(lower_e1, Og1N).nextProbablePrime() */
    sample_in_range(e1, lower_e1, rp->Og1N, ctx);
    BN_generate_prime_ex(e1, BN_num_bits(e1), 0, NULL, NULL, NULL);
    /* Note: BN_generate_prime_ex generates a new prime of the given bit length.
     * To match Java nextProbablePrime(), we need a different approach:
     * find the next prime >= e1. Use BN_add_word + BN_is_prime_ex loop. */

    /* Correct nextProbablePrime: increment e1 until prime */
    if (BN_is_odd(e1) == 0) BN_add_word(e1, 1);  /* make odd */
    while (!BN_is_prime_ex(e1, BN_prime_checks, ctx, NULL)) {
        BN_add_word(e1, 2);
    }
    BN_set_flags(e1, BN_FLG_CONSTTIME);

    /* e2 = sampleInRange(lower_e2, Og2N) */
    sample_in_range(e2, lower_e2, rp->Og2N, ctx);
    BN_set_flags(e2, BN_FLG_CONSTTIME);

    /* S1 = g2^e2 * g1^e1 mod N */
    BN_mod_exp_mont_consttime(tmp1, rp->g2, e2, rp->N, ctx);
    BN_mod_exp_mont_consttime(tmp2, rp->g1, e1, rp->N, ctx);
    BN_mod_mul(S1, tmp1, tmp2, rp->N, ctx);

    /* S2 = (h - s*S1) * e1^(-1) mod phiN */
    BN_mod_mul(tmp1, s_key, S1, rp->phiN, ctx);    /* tmp1 = s*S1 mod phiN */
    BN_mod_sub(tmp2, h, tmp1, rp->phiN, ctx);       /* tmp2 = h - s*S1 mod phiN */
    BN_mod_inverse(e1_inv, e1, rp->phiN, ctx);      /* e1_inv = e1^(-1) mod phiN */
    BN_mod_mul(S2, tmp2, e1_inv, rp->phiN, ctx);    /* S2 = tmp2 * e1_inv mod phiN */

    /* S3 = h - t*S1 - e2*S2 mod phiN */
    BN_mod_mul(tmp1, t_key, S1, rp->phiN, ctx);     /* tmp1 = t*S1 mod phiN */
    BN_mod_mul(tmp2, e2, S2, rp->phiN, ctx);        /* tmp2 = e2*S2 mod phiN */
    BN_mod_sub(tmp3, h, tmp1, rp->phiN, ctx);       /* tmp3 = h - t*S1 mod phiN */
    BN_mod_sub(S3, tmp3, tmp2, rp->phiN, ctx);      /* S3 = tmp3 - e2*S2 mod phiN */

    /* Serialize: sig = S1 || S2 || S3 || message */
    BN_bn2binpad(S1, sig, (int)params->s1_bytes);
    BN_bn2binpad(S2, sig + params->s1_bytes, (int)params->s2_bytes);
    BN_bn2binpad(S3, sig + params->s1_bytes + params->s2_bytes, (int)params->s3_bytes);
    if (msglen > 0) {
        memcpy(sig + params->signature_overhead, msg, (size_t)msglen);
    }
    *siglen = params->signature_overhead + msglen;

    ret = KAZ_SIGN_SUCCESS;

cleanup:
    kaz_secure_zero(hash_buf, params->hash_bytes);
    free(hash_buf);
    BN_clear_free(s_key); BN_clear_free(t_key);
    BN_clear_free(e1); BN_clear_free(e2);
    BN_clear_free(e1_inv);
    BN_free(h); BN_free(S1); BN_free(S2); BN_free(S3);
    BN_free(tmp1); BN_free(tmp2); BN_free(tmp3);
    BN_free(lower_e1); BN_free(lower_e2);
    BN_CTX_free(ctx);
    return ret;
}
```

### Task 6: Rewrite `sign.c` — Verification Function

**Files:**
- Modify: `SIGN/src/internal/sign.c`

**Step 1: Rewrite `kaz_sign_verify_ex()`**

Match Java `KAZSIGNVerifier.verify()`. Input: S1 || S2 || S3 || message. Single verification equation.

```c
int kaz_sign_verify_ex(kaz_sign_level_t level,
                        unsigned char *msg,
                        unsigned long long *msglen,
                        const unsigned char *sig,
                        unsigned long long siglen,
                        const unsigned char *pk)
{
    const kaz_sign_level_params_t *params;
    kaz_runtime_params_t *rp;
    BN_CTX *ctx = NULL;
    BIGNUM *V = NULL, *S1 = NULL, *S2 = NULL, *S3 = NULL, *h = NULL;
    BIGNUM *lhs = NULL, *rhs = NULL;
    BIGNUM *tmp1 = NULL, *tmp2 = NULL, *tmp3 = NULL;
    BIGNUM *g1g2 = NULL;
    unsigned char *hash_buf = NULL;
    const unsigned char *embedded_msg;
    unsigned long long embedded_msglen;
    int ret = KAZ_SIGN_ERROR_VERIFY;

    params = kaz_sign_get_level_params(level);
    if (!params) return KAZ_SIGN_ERROR_INVALID;
    if (!sig || !pk) return KAZ_SIGN_ERROR_INVALID;
    if (siglen < params->signature_overhead) return KAZ_SIGN_ERROR_INVALID;

    ret = kaz_sign_init_level(level);
    if (ret != KAZ_SIGN_SUCCESS) return ret;
    rp = get_runtime_params(level);

    /* Extract embedded message */
    embedded_msg = sig + params->signature_overhead;
    embedded_msglen = siglen - params->signature_overhead;

    /* Hash the embedded message */
    hash_buf = calloc(1, params->hash_bytes);
    if (!hash_buf) return KAZ_SIGN_ERROR_MEMORY;
    ret = kaz_sign_hash_ex(level, embedded_msg, embedded_msglen, hash_buf);
    if (ret != KAZ_SIGN_SUCCESS) { free(hash_buf); return ret; }

    ctx = BN_CTX_new();
    V = BN_new(); S1 = BN_new(); S2 = BN_new(); S3 = BN_new();
    h = BN_new(); lhs = BN_new(); rhs = BN_new();
    tmp1 = BN_new(); tmp2 = BN_new(); tmp3 = BN_new();
    g1g2 = BN_new();
    /* ... NULL checks ... */

    /* Deserialize */
    BN_bin2bn(pk, (int)params->v_bytes, V);
    BN_bin2bn(sig, (int)params->s1_bytes, S1);
    BN_bin2bn(sig + params->s1_bytes, (int)params->s2_bytes, S2);
    BN_bin2bn(sig + params->s1_bytes + params->s2_bytes, (int)params->s3_bytes, S3);
    BN_bin2bn(hash_buf, (int)params->hash_bytes, h);

    /* V = V mod N, S1' = S1 mod N */
    BN_mod(V, V, rp->N, ctx);
    BIGNUM *S1mod = BN_new();
    BN_mod(S1mod, S1, rp->N, ctx);

    /* LHS = V^S1' * S1'^S2 * g2^S3 mod N */
    BN_mod_exp(tmp1, V, S1mod, rp->N, ctx);        /* V^S1' mod N */
    BN_mod_exp(tmp2, S1mod, S2, rp->N, ctx);       /* S1'^S2 mod N */
    BN_mod_exp(tmp3, rp->g2, S3, rp->N, ctx);      /* g2^S3 mod N */
    BN_mod_mul(lhs, tmp1, tmp2, rp->N, ctx);        /* V^S1' * S1'^S2 */
    BN_mod_mul(lhs, lhs, tmp3, rp->N, ctx);         /* * g2^S3 */

    /* RHS = (g1*g2)^h mod N */
    BN_mul(g1g2, rp->g1, rp->g2, ctx);             /* g1*g2 */
    BN_mod_exp(rhs, g1g2, h, rp->N, ctx);           /* (g1*g2)^h mod N */

    /* Compare */
    if (BN_cmp(lhs, rhs) == 0) {
        /* Valid — copy message out */
        if (msg && embedded_msglen > 0) {
            memcpy(msg, embedded_msg, (size_t)embedded_msglen);
        }
        if (msglen) *msglen = embedded_msglen;
        ret = KAZ_SIGN_SUCCESS;
    } else {
        ret = KAZ_SIGN_ERROR_VERIFY;
    }

cleanup:
    kaz_secure_zero(hash_buf, params->hash_bytes);
    free(hash_buf);
    BN_free(V); BN_free(S1); BN_free(S2); BN_free(S3);
    BN_free(S1mod);
    BN_free(h); BN_free(lhs); BN_free(rhs);
    BN_free(tmp1); BN_free(tmp2); BN_free(tmp3);
    BN_free(g1g2);
    BN_CTX_free(ctx);
    return ret;
}
```
